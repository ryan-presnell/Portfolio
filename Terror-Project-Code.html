Donut chart

victim.tab=victim.tab[-1,]
victim.tab$fraction <- victim.tab$`# Attacks` / sum(victim.tab$`# Attacks`)
victim.tab = victim.tab %>% arrange(fraction)

victim.tab$ymax <- cumsum(victim.tab$fraction)


victim.tab$ymin <- c(0, head(victim.tab$ymax, n=-1))


victim.tab$labelPosition <- (victim.tab$ymax + victim.tab$ymin) / 2


victim.tab$label <- paste0(round(victim.tab$`# Attacks`/sum(victim.tab$`# Attacks`)*100,1),"%")


ggplot(victim.tab, aes(ymax=ymax, ymin=ymin, xmax=4, xmin=3, fill=Victim)) +
  geom_rect() +
  geom_text( x=3.5, aes(y=labelPosition, label=label),fill="transparent",color="black",size=4) +
  coord_polar(theta="y",start=90) +
  xlim(c(2, 4)) +
  theme_void() +
  ggtitle("Target Distribution")+
  scale_fill_manual(values = c("#e4e4e4","#c9d4d7","#aec4c9","#92b5bc","#76a5af","#99badd"))+
  theme(plot.title = element_text(hjust = 0.5),text = element_text(size=20,color="black"))
